Git command

常用指令​

git init​

# 初始化 git
git init

git status​

# 查詢 git 狀態
git status

git add​

# 新增檔案
git add <file>

# 新增所有檔案
git add .

git commit​

# 新增檔案
git commit -m '<commit_message>'

# 合併新增檔案(無法新增新的檔案)
git commit -am '<commit_message>'

# 修改最後一次的 commit 紀錄
git commit --amend -m '<new_commit_message>'

# 併入最後一次 commit 紀錄裡
git commit --amend --no-edit

git log​

# 一行顯示 log 命令
git log --oneline --graph

# 查詢作者
git log --oneline --author="<author>"

# 查詢 commit message 有 search commit message
git log --oneline --grep="<search_commit_message>"

# 查詢時間範圍
git log --oneline --since="9am" --until="12am" --after="2023-06-15"

# 查詢所有紀錄 (同 git reflog)
git log -g

gir rm​

# 刪除檔案
git rm <file>

# 移出 git 追蹤
git rm <file> --cached

git checkout​

# 切換分支
git checkout <branch_name>

# 切換並創建分支
git checkout -b <branch_name>

git reset​

# 退回上一動
git reset <commit id>

# 把暫存區的檔案丟掉 (default)
git reset <commit id> --mixed

# 把暫存區和工作目錄的檔案丟掉
git reset <commit id> --hard

# 移動HEAD
git reset <commit id> --soft

# 移動回危險動作前
git reset ORIG_HEAD --hard

git branch​

# 查詢所有分支
git branch

# 新增分支
git branch <new_branch_name>

# 改分支的名字
git branch -m <old_branch_name> <new_branch_name>

# 刪除分支
git branch -d <branch_name>

git merge​

# 合併分支
git merge <branch_name>

# 強制產生新的 commit
git merge <branch_name> --no-ff

git rebase​

# 合併分支
git rebase <branch_name>

# 進入互動模式
git rebase -i <commit_id>

git tag​

# 查詢所有 tag
git tag

# 新增 tag
git tag <tag name>

# 刪除 tag
git tag -d <tag name>

# 附註標籤
git tag -am "<tag_desc>" <tag_name>

git stash​

# 暫存工作目錄
git stash

# 查詢所有暫存
git stash list

# 恢復暫存
git stash apply

# 恢復暫存並刪除暫存
git stash pop <stash id>

# 刪除暫存
git stash drop <stash id>

# 刪除所有暫存
git stash clear

.gitignore​

.gitignore
# 忽略特定檔案檔案 
.env

# 忽略特定目錄 
/node_modules

# 忽略所有附檔名的檔案
*.tmp
